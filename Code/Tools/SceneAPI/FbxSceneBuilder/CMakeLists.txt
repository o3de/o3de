    return()
endif()

ly_add_target(
    NAME FbxSceneBuilder.Static STATIC
    NAMESPACE AZ
    FILES_CMAKE
        fbxscenebuilder_files.cmake
        Platform/${PAL_PLATFORM_NAME}/platform_${PAL_PLATFORM_NAME_LOWERCASE}_files.cmake
    COMPILE_DEFINITIONS
        PRIVATE
            FBX_SCENE_BUILDER_EXPORTS
    INCLUDE_DIRECTORIES
        PUBLIC
            ../..
    BUILD_DEPENDENCIES
        PRIVATE
            AZ::AzCore
            AZ::AzFramework
        PUBLIC
            AZ::AzToolsFramework
            AZ::SceneCore
            AZ::SceneData
            AZ::SDKWrapper
)

ly_add_target(
    NAME FbxSceneBuilder MODULE
    NAMESPACE AZ
    FILES_CMAKE
        fbxscenebuilder_shared_files.cmake
    COMPILE_DEFINITIONS
        PRIVATE
            FBX_SCENE_BUILDER_EXPORTS
    INCLUDE_DIRECTORIES
        PUBLIC
            ../..
    BUILD_DEPENDENCIES
        PUBLIC
            AZ::FbxSceneBuilder.Static
        PRIVATE
            AZ::AzCore
)

ly_add_dependencies(AssetBuilder AZ::FbxSceneBuilder)
